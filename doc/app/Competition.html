def result_headings()
        case self.sport
                when "basketball"
                        ["Wins", "Losses", "Pct"]
                when "soccer"
                        ["Wins", "Losses", "Draws", "Points", "GF", "GA", "GD"]
                else
                        ["Wins", "Losses"]
                end
